<script>
(function () {
  "use strict";

  var root = document.getElementById("cmp-recent-posts");
  if (!root) return;

  var list = root.querySelector(".cmp-recent-list");
  var status = root.querySelector(".cmp-recent-status");
  var updated = root.querySelector(".cmp-recent-updated");

  function makeLink(className, text, href) {
    var link = document.createElement("a");
    link.className = className;
    link.textContent = text;
    link.href = href;
    link.target = "_blank";
    link.rel = "noopener noreferrer";
    return link;
  }

  function formatPostDate(value) {
    var date = new Date(value);
    if (Number.isNaN(date.getTime())) return "Date unavailable";
    return new Intl.DateTimeFormat("en-GB", {
      day: "numeric",
      month: "short",
      year: "numeric"
    }).format(date);
  }

  function formatUpdated(value) {
    var date = new Date(value);
    if (Number.isNaN(date.getTime())) return "";
    return new Intl.DateTimeFormat("en-GB", {
      day: "numeric",
      month: "long",
      year: "numeric",
      hour: "2-digit",
      minute: "2-digit",
      timeZone: "Europe/London",
      timeZoneName: "short"
    }).format(date);
  }

  window.__cmpRenderBlaugustRecentPosts = function (data) {
    list.replaceChildren();
    var posts = data && Array.isArray(data.posts) ? data.posts : [];

    if (!posts.length) {
      status.textContent = "No recent posts are available yet. The feed updater may still be completing its first run.";
      status.hidden = false;
      list.hidden = true;
      updated.hidden = true;
      return;
    }

    posts.forEach(function (post) {
      var item = document.createElement("li");
      item.appendChild(makeLink("cmp-recent-title", post.post_title, post.post_url));

      var meta = document.createElement("div");
      meta.className = "cmp-recent-meta";
      meta.appendChild(makeLink("cmp-recent-blog", post.blog_title, post.blog_url));
      meta.appendChild(document.createTextNode(" · "));

      var time = document.createElement("time");
      time.dateTime = post.published;
      time.textContent = formatPostDate(post.published);
      meta.appendChild(time);
      item.appendChild(meta);
      list.appendChild(item);
    });

    status.hidden = true;
    list.hidden = false;

    var refreshed = formatUpdated(data.generated_at);
    if (refreshed) {
      updated.textContent = "Feed list last refreshed " + refreshed + ".";
      updated.hidden = false;
    }
  };

  if (window.__blaugustRecentPostsData) {
    window.__cmpRenderBlaugustRecentPosts(window.__blaugustRecentPostsData);
    return;
  }

  var script = document.createElement("script");
  script.src = root.getAttribute("data-source") + "?v=" + Date.now();
  script.async = true;
  script.onerror = function () {
    status.textContent = "The recent-post list could not be loaded at present.";
  };
  document.head.appendChild(script);
})();
</script>
